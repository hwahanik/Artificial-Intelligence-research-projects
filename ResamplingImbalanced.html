from sklearn.datasets import make_classification
import numpy as np
import pandas as pd
import os
import seaborn as sns; sns.set()
from sklearn.decomposition import PCA
import matplotlib.pyplot as plt


X,y = make_classification(n_classes = 2, class_sep=1.5, weights=[0.9, 0.1],
    n_informative=3, n_redundant=1, flip_y=0,
    n_features=20, n_clusters_per_class=1,
    n_samples=100, random_state=10)

df = pd.DataFrame(X)
df['target'] = y
sns.distplot(y, kde = False, rug = True)

pca = PCA(n_components = 2)
X_2D_decomposed = pd.DataFrame(pca.fit_transform(X))
X_2D_decomposed['target'] = y
X_2D_decomposed.columns = ['Dim 1', 'Dim 2', 'label']

from imblearn.over_sampling import RandomOverSampler

ros = RandomOverSampler(return_indices = True)
X_ros, y_ros, id_ros = ros.fit_sample(X, y)
print('new indexes:', id_ros)

print(X_ros.shape[0] - X.shape[0], 'new random picked points')
