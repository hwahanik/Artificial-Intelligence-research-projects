# Calculation of the VaR using the variance-covariance approach and the historical approach
# We simulate the stock prices using the brownian motion approx for stock movements.

import matplotlib.pyplot as plt
import numpy as np
import pandas as pd
import matplotlib.mlab as mlab
from scipy.stats import norm
from tabulate import tabulate
import os

def stock_price(T, mu, sigma, S0, dt):
    # Simulation of brownian motion for stock price
    N = round(T/dt)
    t = np.linspace(0, T, N)
    W = np.random.standard_normal(size = N) 
    W = np.cumsum(W)*np.sqrt(dt) ### standard brownian motion ###
    X = (mu-0.5*sigma**2)*t + sigma*W 
    S = S0*np.exp(X) ### geometric brownian motion ###
    return t, S

T = 30
mu = 0.1
sigma = 0.2
dt = 0.1

S_ask, S_close = 20 , 25
t_ask, S_ask = stock_price(T, mu, sigma, S_ask, dt)
t_close, S_close = stock_price(T, mu, sigma, S_close, dt)

plt.plot(t_ask, S_ask)
plt.plot(t_close, S_close)
plt.show()
